// Функция для удаления всех данных из localStorage
    function clearLocalData() {
        localStorage.clear(); // Удаляет все данные из localStorage
        console.log('[DEBUG]: Все локальные данные удалены.');
        
        // Дополнительно можно перезагрузить страницу для применения изменений
        window.location.reload();
    }

    // Привязка функции к кнопке (если хотите запускать по клику)
    document.addEventListener('DOMContentLoaded', () => {
        const clearBtn = document.getElementById('clearLocalDataBtn');
        if (clearBtn) {
            clearBtn.addEventListener('click', clearLocalData);
        } else {
            console.warn('[WARN]: Кнопка с ID "clearLocalDataBtn" не найдена. Запустите clearLocalData() вручную в консоли.');
        }
    });
